\section{Failure-mode catalog}
\label{app:failures}

\begin{itemize}\itemsep0pt
\item \textbf{Class A -- wrong-form.} Agent submits a textbook law that
      the shift breaks (e.g.\ submits $F=-kx$ on $\gamma$-1-1 where
      the true law is $F = -kx(1+\eta\tanh(x/x_0))$). Scores zero on
      the OOD sub-grid by construction.
\item \textbf{Class B -- right-name-wrong-binding.} Agent submits the
      correct symbolic law but the \texttt{def f(...)} predictor's
      argument names do not match the harness's grid input vocabulary.
      The two-stage scorer marks dimensional pass, numerical zero.
      Common on Claude variants.
\item \textbf{Class C -- partial-form.} Agent submits a predictor that
      matches the in-grid (a) sub-grid but fails on (b)/(c). Drives
      GPT-5.4's non-trivial $\gamma$ scores when present.
\item \textbf{Class D -- budget-out.} Agent never calls
      \texttt{submit\_answer}; \texttt{terminated\_by=budget}.
      Empty submission, $S=0$.
\end{itemize}

\section{Reproducibility}
\label{app:repro}

\paragraph{Environment.} Python $\geq 3.11$, requirements pinned in
\texttt{pyproject.toml}.

\paragraph{Commands.}
\begin{verbatim}
# Ceiling sweep (48 pairs, oracle agent)
python -m mirrorlab.runners.ceiling_sweep \
    --out docs/ceiling-data.json --seed 0

# 5-model x 12-cell honest sweep
python -m mirrorlab.runners.sprint4_full_sweep \
    --models claude-opus-4.6 claude-sonnet-4.5 \
             gemini-3.1-pro-preview gpt-4.1-20250414 \
             gpt-5.4-20260305 \
    --out docs/sprint4-sweep-data-final.json --seed 0

# Lookup attacker (Sprint 3.5 protocol)
python -m mirrorlab.runners.sprint35_pilot \
    --attacker-only --out docs/sprint35-pilot-data.json

# Figures
python -m mirrorlab.reports.figures
\end{verbatim}

The proxy endpoint and API key live in the project's local secret store;
they are not committed.
